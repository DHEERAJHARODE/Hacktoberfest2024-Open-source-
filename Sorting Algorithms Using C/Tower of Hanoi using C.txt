#include <stdio.h>
 
void towers(int, char, char, char);
 
int main()
{
    int number;
 
    printf("Enter the number of total disks : ");
    scanf("%d", &number);
    printf("The sequence of moves involved in the Tower of Hanoi are :\n");
    towers(number, 'A', 'C', 'B');
    return 0;
}
void towers(int number, char start, char final, char middle)
{
    
    if (number == 1)
    {
        printf("\n Move disk 1 from peg %c to peg %c", start, final);
        return;
    }
 
    towers(number - 1, start, middle, final);
    printf("\n Move disk %d from peg %c to peg %c", number, start, final);
    towers(number - 1, middle, final, start);
}
